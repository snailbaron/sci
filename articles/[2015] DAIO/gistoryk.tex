\medskip\remarka{ 1} В ряде случаев результаты, полученные для задачи
\mbox{$2$-PSP}, могут быть использованы для $2$-CPSP. Любое
допустимое решение $2$-PSP является допустимым решением $2$-CPSP, а
величина $2W(T^*)$ является нижней (верхней) оценкой для решения
$2$-CPSP$_{\min}$ ($2$-CPSP$_{\max}$). Любой алгоритм $A$ решения
задачи $2$-PSP$_{\min}$ ($2$-PSP$_{\max}$), для которого доказана
оценка вида $W(P^A) \leq \alpha \cdot 2W(T^*)$ ($W(P^A) \geq \alpha
\cdot 2W(T^*)$), является $\alpha$-приближённым алгоритмом для
задачи $2$-CPSP$_{\min}$ ($2$-CPSP$_{\max}$); в~качестве решения
$2$-CPSP возьмём решение $2$-PSP, полученное алгоритмом $A$.
Действительно, для задачи на минимум из неравенства $W(H^*) \geq
2W(T^*)$ следует, что  $$ W(H^A) = W(P^A) \leq \alpha \cdot 2W(T^*)
\leq \alpha \cdot W(H^*).$$ Аналогично, для задачи на максимум имеем
$ W(H^A) \geq \alpha \cdot W(H^*)$.

\begin{theorem}
\label{theorem_approx_ratio_min} Предположим, что в алгоритме
$A_{\min}\{\overline{1,q}\}$ используется полиномиальный
приближённый алгоритм решения {\rm TSP}$_{\min}\{\overline{1,q}\}$ с
гарантированной оценкой точности $\Delta$. Тогда в полном
$n$-вершинном графе с весами рёбер из целочисленного сегмента
$\{\overline{1,q} \}$ алгоритм $A_{\min}\{\overline{1,q}\}$ находит
приближённое решение задачи $2$-{\rm CPSP}$_{\min}\{\overline{1,q}
\}$ с гарантированной оценкой точности $\rho$ в среднем
$$
\rho \leq
\left\{
  \begin{array}{ll}
    \frac{(1 + p)\Delta + (1-p)q}{2}
       &\mbox{при } n \geq n_0, \\
    \frac{(1 + p)\Delta + (1-p)q}{2} + \varepsilon & \mbox{при } n <
    n_0,
  \end{array}
\right.
$$
где $ \varepsilon \leq \frac{q}{2n}, \quad n_0 = \left\{
  \begin{array}{ll}
    \min\left\{ \frac{q+5}{1-p}, \frac{7}{1 + (\Delta - q)p}  \right\}, &$если$ \ 1 + (\Delta - q)p > 0,\\
    \frac{q+5}{1-p} &$в противном случае$.
  \end{array}
\right.
$
\end{theorem}
\proof Текст доказательства. \thmp


\begin{sled}
\label{sled_approx_ratio_min_1_2} Используя $7/6$-приближ\"енный
алгоритм  решения задачи {\rm TSP}$_{\min}\{1,2\}$, алгоритм
$A_{\min}\{\overline{1,q} \}$ ($q=2$) строит решение задачи $2$-{\rm
CPSP}$_{\min}\{1,2\}$ с гарантированной оценкой точности в среднем
не хуже, чем $(19-5p)/{12}$, при $n \geq \frac{7}{1 -
\frac{5}{6}p}$.
\end{sled}
